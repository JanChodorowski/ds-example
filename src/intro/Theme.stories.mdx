## Colors

<DocThemeProvider>

      <PaletteItem

        name={c}
        description={colors[c]}
        code={c}
        key={c}
        sampleFrame={
          (colors[c].includes('rgba') || colors[c].includes('fff')) && 'rgba'

      />
